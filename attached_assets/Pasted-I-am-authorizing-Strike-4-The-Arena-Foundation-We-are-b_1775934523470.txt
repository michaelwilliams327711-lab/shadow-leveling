import { Router } from "express";
import { db } from "@workspace/db";
import { bossesTable, bossDamageLogTable } from "@workspace/db";
import { eq, and } from "drizzle-orm";
import { getOrCreateCharacter } from "./character.js";

const router = Router();

// Get all bosses for the character
router.get("/bosses", async (req, res) => {
  try {
    const char = await getOrCreateCharacter();
    const bosses = await db.select().from(bossesTable).where(eq(bossesTable.characterId, char.id));
    res.json(bosses);
  } catch (err) {
    res.status(500).json({ error: "Internal server error" });
  }
});

export default router;
