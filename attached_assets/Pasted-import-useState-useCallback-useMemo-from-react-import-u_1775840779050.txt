// TugBar: center=50%; virtue score pushes right; vice score pushes left
function TugBar({ viceScore, virtueScore }: { viceScore: number; virtueScore: number }) {
  const total = viceScore + virtueScore;
  const raw = total === 0 ? 50 : 50 + ((virtueScore - viceScore) / Math.max(total, 1)) * 50;
  const pct = Math.min(95, Math.max(5, raw)); // Clamp visually
  
  return (
    <div className="relative h-3 w-full rounded-full bg-white/5 overflow-hidden">
      {/* Dynamic Tug Layer */}
      <div 
        className="absolute inset-y-0 left-0 rounded-full tug-bounce"
        style={{ 
          width: `${pct}%`,
          background: pct >= 50 
            ? "linear-gradient(to right, #7f1d1d, #a16207)" // Red to Amber (Virtue winning area)
            : "linear-gradient(to right, #7f1d1d, #6b21a8)" // Red to Purple (Vice winning area)
        }} 
      />
      {/* Directional Labels (Audit U-003) */}
      <div className="absolute inset-0 flex items-center justify-between px-2 text-[8px] font-black tracking-widest text-white/40 uppercase pointer-events-none">
        <span>SIN ←</span>
        <span>→ VIRTUE</span>
      </div>
      {/* Pointer/Center Marker */}
      <div 
        className="absolute inset-y-0 w-1 -translate-x-1/2 bg-white rounded-full shadow-[0_0_6px_rgba(255,255,255,0.8)] tug-bounce"
        style={{ left: `${pct}%` }} 
      />
    </div>
  );
}

// Global Battlefield Clash Component (Omega Audit)
function GlobalBattlefield({ powers }: { powers: CelestialPower[] }) {
  // ── Calculate Global Balance ──
  const totals = useMemo(() => powers.reduce((acc, p) => ({
    vice: acc.vice + p.viceScore,
    virtue: acc.virtue + p.virtueScore
  }), { vice: 0, virtue: 0 }), [powers]);

  const totalPoints = totals.vice + totals.virtue;
  const virtueGloballyWinning = totals.virtue > totals.vice;
  
  // No battle yet
  if (totalPoints === 0) {
    return (
      <div className="relative flex h-48 md:h-64 overflow-hidden border-b border-white/10 bg-black">
        <div className="absolute inset-0 z-0 flex">
          <div className="flex-1" style={{ background: `url(${sinsImg}) center/cover, #2a0340` }} />
          <div className="flex-1" style={{ background: `url(${virtuesImg}) center/cover, #1d4e89` }} />
        </div>
        <div className="absolute inset-0 z-10 bg-black/60" />
        <div className="relative z-20 w-full flex items-center justify-center">
          <h2 className="font-display text-2xl font-black tracking-widest text-muted-foreground uppercase">Tide is Still</h2>
        </div>
      </div>
    );
  }

  // ── Animated Battlefield Clash ──
  return (
    <div className="relative flex h-48 md:h-64 overflow-hidden border-b border-white/10 bg-black celestial-battlefield">
      
      {/* Virtue Side (Light blue/gold particles, expansive glow) */}
      <div className={`absolute inset-y-0 right-0 z-10 flex items-center justify-center overflow-hidden transition-all duration-1000 ease-out ${virtueGloballyWinning ? 'winner-virtue' : 'loser-virtue'}`}
        style={{ 
          background: `url(${virtuesImg}) center/cover fixed, linear-gradient(135deg,#a16207 0%,#065f46 100%)`,
          animation: virtueGloballyWinning ? 'clash-virtue-win 1.5s forwards ease-out' : 'clash-virtue-retreat 1.5s forwards ease-in'
        }}
      >
        <div className="absolute inset-0 bg-black/40" style={{ animation: virtueGloballyWinning ? 'virtueGlow 5s infinite' : 'none' }} />
        {/* Driving Particles */}
        {virtueGloballyWinning && (
          <div className="absolute inset-0 z-10 arise-mana" style={{ animation: 'battlefield-particles-virtue-win 3s linear infinite' }} />
        )}
        <div className="relative z-20 text-center px-4">
          <p className="text-xs tracking-[0.5em] text-amber-400 uppercase font-display mb-1">The Seven</p>
          <h2 className="font-display text-3xl md:text-4xl font-black tracking-widest text-amber-200 drop-shadow-lg">VIRTUES</h2>
          {virtueGloballyWinning && <span className="text-[10px] tracking-widest text-white/70 uppercase">→ DRIVING THE TIDE ←</span>}
        </div>
      </div>

      {/* Central Separator */}
      <div className="absolute left-1/2 top-0 bottom-0 -translate-x-1/2 w-px bg-white/20 z-20" />
      <div className="absolute left-1/2 top-1/2 -translate-x-1/2 -translate-y-1/2 z-30">
        <div className="flex h-12 w-12 items-center justify-center rounded-full border border-white/30 bg-black/80 shadow-[0_0_20px_rgba(255,255,255,0.15)]">
          <LucideScale className="h-6 w-6 text-white" />
        </div>
      </div>

      {/* Sin Side (Dark red/purple smoke, retreating glow) */}
      <div className={`absolute inset-y-0 left-0 z-10 flex items-center justify-center overflow-hidden transition-all duration-1000 ease-out ${!virtueGloballyWinning ? 'winner-sin' : 'loser-sin'}`}
        style={{ 
          background: `url(${sinsImg}) center/cover fixed, linear-gradient(135deg,#3b0764 0%,#7f1d1d 100%)`,
          animation: !virtueGloballyWinning ? 'clash-sin-win 1.5s forwards ease-out' : 'clash-sin-retreat 1.5s forwards ease-in'
        }}
      >
        <div className="absolute inset-0 bg-black/40" style={{ animation: !virtueGloballyWinning ? 'sinGlow 5s infinite' : 'none' }} />
        {/* Disappearing Smoke/Ember Particles if losing */}
        {virtueGloballyWinning && (
          <div className="absolute inset-0 z-10 corruption-smoke" style={{ opacity: 0.3, animation: 'smoke-retreat-fade 2s forwards' }} />
        )}
        <div className="relative z-20 text-center px-4">
          <p className="text-xs tracking-[0.5em] text-red-400 uppercase font-display mb-1">The Seven</p>
          <h2 className="font-display text-3xl md:text-4xl font-black tracking-widest text-red-300 drop-shadow-lg">SINS</h2>
          {!virtueGloballyWinning && <span className="text-[10px] tracking-widest text-white/70 uppercase">← DRIVING THE TIDE →</span>}
        </div>
      </div>

    </div>
  );
}
